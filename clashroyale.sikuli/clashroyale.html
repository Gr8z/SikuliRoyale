def scroll_down():
    # setting begin - end
    x1, y1 = (r.x + 230, r.y + 700)
    start = Location(x1, y1)
    # moving up
    stepY = 100
    run = start
    mouseMove(start); wait(0.5)
    mouseDown(Button.LEFT); wait(0.5)
    for i in range(4):
        run = run.above(stepY * i)
        mouseMove(run)
    mouseUp()
